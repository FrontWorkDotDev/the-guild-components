import { fetchPackageInfo, Tabs, RemoteContent } from '@theguild/components'
import { compileMdx } from '@theguild/components/nextra'
import { defaultRemarkPlugins } from '@theguild/components/next.config'

export async  function PackageApiDocs () {
  const PACKAGE_NAME = '@envelop/core'
  const { readme } = await fetchPackageInfo(PACKAGE_NAME)
  const { result } = await compileMdx(
    `
\`\`\`sh npm2yarn
npm i ${PACKAGE_NAME}
\`\`\`
${readme}`,
    {
      defaultShowCopyCode: true,
      mdxOptions: {
        remarkPlugins: defaultRemarkPlugins
      }
    }
  )
  return <RemoteContent compiledSource={result} components={{ $Tabs: Tabs }} />
}
